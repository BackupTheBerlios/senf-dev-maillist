#include <senf/Scheduler/ClockService.hh>
#include <senf/Utils/membind.hh>
#include <senf/Scheduler/Scheduler.hh>


class IpmuxSock {
public:

    void timeout(){
        std::cout<<senf::ClockService::in_milliseconds( senf::ClockService::now())<<std::endl;
      senf::Scheduler::instance().timeout( senf::Scheduler::instance().eventTime() + senf::ClockService::milliseconds(11), senf::membind(&IpmuxSock::timeout, this));    
    }
};



int main( int argc, char **argv){

    IpmuxSock t;
    
    t.timeout();

    senf::Scheduler::instance().process();

}
